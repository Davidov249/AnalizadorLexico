DROP ::= DROP DROPP FINSENTENCIA

DROPP ::= DROPPT
        | DROPPD
        | DROPPU //Cambio de Login por User
        | DROPPV
        | DROPPI

DROPPT ::= TABLE IFEXT

IFEXT ::= IF EXISTS IFEXT1
        | IFEXT1

IFEXT1 ::= ID IFEXT2

IFEXT2 ::= "." ID IFEXT3
        | IFEXT3

IFEXT3 ::= "." ID IFEXT4
        | IFEXT4

IFEXT4 ::= "," IFEXT1 
        | VACIO

DROPPU ::= USER IFEXU

IFEXU ::= IF EXISTS IFEXU1
	| IFEXU1

IFEXU1 ::= ID  

DROPPD ::= DATABASE IFEXD

IFEXD ::= IF EXISTS IFEXD1
        | IFEXD1

IFEXD1 ::= ID IFEXD2 

IFEXD2 ::= "," IFEXD1
	| VACIO

DROPPV ::= VIEW IFEXV

IFEXV ::= IF EXISTS IFEXV1
        | IFEXV1

IFEXV1 ::= ID IFEXV2
        | IFEXV2
        
IFEXV2 ::= "." ID IFEXV3
        | IFEXV3

IFEXV3 ::= "," IFEXV1
	| VACIO

DROPPI ::= INDEX IFEXI

IFEXI ::= IF EXISTS IFEXI1
        | IFEXI1

IFEXI1 ::= ID DROPI1

DROPI1 ::= ON OBJECT DROPI3
	| "." ID DROPI2 

DROPI2 ::= "." ID DROPI3
	| DROPI3

DROPI3 ::= "," IFEXT1
	| VACIO

/* FACTORIZACION
IFEXI1 ::= DROPRELIND
        |  DROPBACKIND

DROPRELIND ::= DROPRELIND1 DROPRELIND2

DROPRELIND2 ::= "," IFEXI1
                | VACIO

DROPBACKIND ::= DROPBACKIND1 DROPBACKIND3

DROPBACKIND3 ::= "," IFEXT1
                | VACIO

DROPRELIND1 ::= ID ON OBJECT 

DROPBACKIND1 ::= ID DROPBACKIND2
                | DROPBACKIND2

DROPBACKIND2 ::= ID "." ID
