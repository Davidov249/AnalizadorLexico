DROPP ::= DROPPT
        | DROPPD
        | DROPPL
        | DROPPV
        | DROPPI

DROPPT ::= DROP TABLE IFEXT

IFEXT ::= IF EXISTS IFEXT1
        | IFEXT1

IFEXT1 ::= ID IFEXT2

IFEXT2 ::= "." ID IFEXT3
        | IFEXT3

IFEXT3 ::= "." ID IFEXT4
        | IFEXT4

IFEXT4 ::= "," DROPPT 
        | ";" INICIO

DROPPL ::= DROP LOGIN ID ";" INICIO

DROPPD ::= DROP DATABASE IFEXD

IFEXD ::= IF EXISTS
        | IFEXD1

IFEXD1 ::= ID "," DROPPD
        | ";" INICIO

DROPPV ::= DROP VIEW IFEXV

IFEXV ::= IF EXISTS IFEXV1
        | IFEXV1

IFEXV1 ::= ID "." IFEXV2
        | IFEXV2
        
IFEXV2 ::= ID "," DROPPV
        | ";" INICIO

DROPPI ::= DROP INDEX IFEXI

IFEXI ::= IF EXISTS IFEXI1
        | IFEXI1

IFEXI1 ::= DROPRELIND
        |  DROPBACKIND

DROPRELIND ::= DROPRELIND1 DROPRELIND2

DROPRELIND2 ::= "," DROPRELIND
                | ";" INICIO

DROPBACKIND ::= DROPBACKIND1 DROPBACKIND3

DROPBACKIND3 ::= "," DROPBACKIND 
                | ";" INICIO

DROPRELIND1 ::= ID ON OBJECT 

DROPBACKIND1 ::= ID DROPBACKIND2
                | DROPBACKIND2

DROPBACKIND2 ::= ID "." ID
