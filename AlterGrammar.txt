ALTERP ::= ALTER ALTERP1

ALTERP1 ::= ALTERV
        |   ALTERT
        |   ALTERL //ALTER LOGIN NO SE PUEDE HACER
        |   ALTERD
        |   ALTERI

ALTERV ::= VIEW ALTERV1 ALTERV3 AS SELECT_STATEMENT ALTERV5

ALTERV1 ::= ID ALTERV2

ALTERV2 ::= "." ID
        |   VACIO

ALTERV3 ::= "(" ID ALTERV4
        |   VACIO

ALTERV4 ::= "," ID  ALTERV4
        |   ID ")"

ALTERV5 ::= WITH CHECK OPTION ";" INICIO
        |   ";" INICIO
